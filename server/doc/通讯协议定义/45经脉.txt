%- 角色信息协议定义：
%-     分类号:45
%-     描述: 玩家经脉 
%- 100 提升成功
%- 101 位置错误
%- 102 经脉已经处于修炼状态,还处于冷却时间(还没有完成)
%- 103 经脉模板表没有数据
%- 104 经脉已满级
%- 105 角色已经在修炼了，不能同时修炼两个
%- 106 角色没有在修炼
%- 107 筋骨已满级
%- 108 不存在该经脉类型
%- 109 经脉等级超过玩家等级
%- 110 提升根骨失败
%- 111 错误的参数值
%- 112 修为需要的经脉1、经脉2等级不足
%- 113 修为等级是否满级
%------------------------------------------
%Protocol: 45001 获取玩家经脉/根骨信息
%------------------------------------------  
c>>s: 
s>>c:
    int:32 TriTime 玩家触发时间
    int:16 CostTime 等待时间
    int:8  State 状态
    array { Mer1List 经脉1列表 
   int:32  MerId 经脉Id
   }
    array { Mer2List 经脉2列表 
   int:32  MerId 经脉Id 
   }
      array { BonesList 根骨列表 
   int:8  BonesLv 根骨等级 
   }
end 

%------------------------------------------
%Protocol: 45002 提升经脉
%------------------------------------------  
c>>s:
   int:8 MerType 筋脉类型(1.经脉1 2.经脉2)
s>>c:
   int:8 Result 操作结果 
end 

%------------------------------------------
%Protocol: 45003 提升筋骨
%------------------------------------------  
c>>s: 
   int:8 MerType 经脉类型(任脉，督脉，带脉。。。)
   int:8 IfProtect 是否使用根骨保护符 
s>>c:
   int:8 Result 操作结果 
end 

%------------------------------------------
%Protocol: 45004  等待一定时间后完成经脉1提升
%------------------------------------------  
c>>s:
s>>c:
   int:8 Result 操作结果 
end 

%------------------------------------------
%Protocol: 45005 加速经脉1修炼
%------------------------------------------  
c>>s: int:8 CostNum 消耗的加速符
s>>c:
   int:8 Result 操作结果 
   int:8 State  经脉状态
   int:16 CostTime 等待时间
end 

%------------------------------------------
%Protocol: 45006 按类型获取经脉2/根骨信息
%------------------------------------------
c>>s:
    int:8  Type 获取信息类型(1.经脉1 2.经脉2 3.根骨)
s>>c:
     array { List 结果列表
   int:32 ItemLId 对应属性的标识  
   }
end 

%------------------------------------------
%Protocol: 45007 提升修为
%------------------------------------------
c>>s:
s>>c:
   int:8 Result 操作结果 
end  

%------------------------------------------
%Protocol: 45008 获取经脉1信息
%------------------------------------------
c>>s:
s>>c:
   int:32 TriTime 玩家触发时间
   int:16 CostTime 等待时间
   int:8  State 状态
    array { Mer1List 经脉1列表 
   int:32  MerId 经脉等级 
   }
end
