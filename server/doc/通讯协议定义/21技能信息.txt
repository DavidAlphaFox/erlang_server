%-分类号:21
%-描述:技能信息

%------------------------------------------
%Protocol: 21000 获取技能列表 
%------------------------------------------
c >> s:
s >> c:
    array { SkillList     技能列表
        int:8 SkillId     技能ID
        int:8 Level       技能等级
    }
end

%------------------------------------------
%Protocol: 21001 学习技能 
%------------------------------------------
c >> s:
    int:8  SkillId 技能ID
s >> c:
    int:8 StCode 0失败; 1成功; 2铜币不足; 3人物等级不足; 4失败XX不足; 5前面所学技能等级不足; 6已学
    int:8 SkillId  新学的技能ID,学习成功后发送
end

%------------------------------------------
%Protocol: 21002 升级技能 
%------------------------------------------
c >> s:
    int:8  SkillId 技能ID
s >> c:
    int:8 StCode 0失败;1成功; 2铜币不足; 3等级不足; 4失败XX不足; 5:未学习
    int:8 SkillId  新学的技能ID,成功后发送
    int:8 SkillLv  新学的技能等级,成功后发送
end


%------------------------------------------
%Protocol: 21003 使用技能
%------------------------------------------
c >> s:
	int:8   SkillId 技能ID
	int:8	技能动作
	int:8	攻击类型(1-人，2-怪)
	int:	攻击目标（64 / 32）
	
s >> c:
    int:8   StCode 0失败;1成功;
    int:8	技能ID
    
end

%------------------------------------------
%Protocol: 21004 技能洗点
%------------------------------------------
c >> s:	
成功:
s >> c:
    int:8   StCode 结果码 100成功
    int:8	SkillPointUsed  已用技能点
	int:8	TotalSkillPoint 总技能点
	array{NewSkillList 新技能列表
		int:8 SkillId 技能id
		int:8 SkillLv 技能等级
	}
失败:
s >> c:
	int:8   StCode 结果码 1.元宝不足 2.系统繁忙
end
