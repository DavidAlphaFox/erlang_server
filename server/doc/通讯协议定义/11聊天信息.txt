%- 聊天信息协议定义：
%-     分类号:11
%- 返回码定义：
%-     0: 未定义错误  1：成功  2: 禁言 3:黑名单 4:不在线 5:发言太快
%-
%------------------------------------------
%Protocol: 11000 聊天信息
%------------------------------------------
s >> c:
 	int:64   Uid      发送方用户id
 	string   Name     发送方名称
 	int:8    Type     消息类型: 1.世界、2.场景、3.联盟 4:私聊
 	string   Content  内容 
 	int:8		Vip_Level vip类型
 	int:8 ShowState 1:普通聊天 0：展示 2:小喇叭广播
end

%------------------------------------------
%Protocol: 11001 发送世界信息
%------------------------------------------
c >> s:
	int:8 1:普通聊天 0:道具展示 2：小喇叭广播
 	string  Content  内容(UTF8)
s >> c:
 	int:8 Result 错误类型
end

%------------------------------------------
%Protocol: 11002 发送阵营信息
%------------------------------------------
c >> s:
 	string  Content  内容(UTF8)
 	int:8 1:普通聊天 0:道具展示
s >> c:
 	int:8 Result 错误类型
end

%------------------------------------------
%Protocol: 11003 发送帮派信息
%------------------------------------------
c >> s:
 	string  Content  内容(UTF8)
 	int:8 1:普通聊天 0:道具展示
s >> c:
 	int:8 Result 错误类型
end

%------------------------------------------
%Protocol: 11004 发送私聊信息
%------------------------------------------
c >> s:
 	int:64  PeerId   接受方用户id
 	string  Content  内容(UTF8)
 	int:8 1:普通聊天 0:道具展示
s >> c:
 	int:8 Result 错误类型
end

%------------------------------------------
%Protocol: 11005 GM指令
%------------------------------------------
c >> s:
	int:8  Type    内容类型-0为普通指令；1为代码
    string Content 内容  
s >> c:
    int:8 Result 错误类型
end

%------------------------------------------
%Protocol: 11006 获取小喇叭数量
%------------------------------------------
c >> s
s >> c:
		int:16 HornNum 小喇叭数量
		
%------------------------------------------
%Protocol: 11007 获取最近联系人列表
%------------------------------------------
c >> s
s >> c:
		array
		{
			int:32 Uid  玩家UId
			string Name 玩家名字
		}

%------------------------------------------
%Protocol: 11010 系统信息/广播
%------------------------------------------
s >> c:
 	int:8    Type     消息类型: 0.普通系统提示面板消息 1.1级公告，会在跑马灯出现;2.2级公告，会在系统公告频道出现，红字。3.3级公告，会在系统频道出现，普通字。
 	int:32   Id       提示码 
end


%------------------------------------------
%Protocol: 11021 消息提示系统公告
%------------------------------------------
s >> c:
    int:32	消息ID
    array{	----参数列表
    	int:8 	参数类型    (1-int:8,2-int:16,3-int:32,4-int:64,5-string)
    	type	参数值。    (根据参数类型确定)   
    }
end


%------------------------------------------
%Protocol: 11099 调试信息
%------------------------------------------
c >> s:
    string Content 内容(最大长度为)
end
