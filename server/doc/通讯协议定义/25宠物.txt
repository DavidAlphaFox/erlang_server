%- 宠物议定义：
%- 分类号:25
%------------------------------------------
%Protocol: 25001 获取宠物信息
%------------------------------------------
c >> s:
s >> c:
  int:8 Result 返回码:0 失败, 1成功, 2没有宠物
	int:16 成长上限
	int:16 当前成长值
	int:16 成长进度
	int:16 成长总进度
	int:16 资质上限
	int:16 当前资质
	int:16 资质进度
	int:16 资质总进度
	int:32 战力
	int:8  品级
	int:32 普攻
	int:16 命中
	int:16 暴击
	int:8  属攻类型:1仙攻,2魔攻,3妖攻
	int:32 属攻
	int:8 技能槽数
	array(
    int:8 插槽上的技能类型id
    int:8  技能等级
  )
  array(
    int:16  外观id
    int:32  过期时间(秒数)
  )
end

%------------------------------------------
%Protocol: 25002 宠物展示
%------------------------------------------
c >> s:
s >> c:
 	int:8 Result 返回码:0 失败, 1成功
end

%------------------------------------------
%Protocol: 25003 宠物名称更改
%------------------------------------------
c >> s:
 	string  Name  新名字(UTF8)
s >> c:
 	int:8 Result 返回码:0 失败, 1成功, 2名字长度错误, 3含特殊字符/敏感词, 4没有宠物
 	string  Name  新名字(UTF8)
end

%------------------------------------------
%Protocol: 25004 宠物休息/参战
%------------------------------------------
c >> s:
 	int:8 Flag  0:休息,1:参战
s >> c:
 	int:8 Result 返回码:0 失败, 1成功, 2没有宠物,3传入参数错误
 	int:8 Flag  0:休息,1:参战
end

%------------------------------------------
%Protocol: 25005 洗髓
%------------------------------------------
c >> s:
 	int:8  AtrrAttack   属攻类型:1仙攻,2魔攻,3妖攻
s >> c:
 	int:8 Result 返回码:0 失败, 1成功, 2没有宠物,3传入参数错误
 	int:8  AtrrAttack   属攻类型:1仙攻,2魔攻,3妖攻
end

%------------------------------------------
%Protocol: 25006 幻化
%------------------------------------------
c >> s:
	int:16  Type	新幻化类型
s >> c:
    int:8 Result 返回码:0 失败, 1成功, 2没有宠物,3幻化类型不存在,4幻化类型和当前类型相同
    int:16  Type	新幻化类型
end

%------------------------------------------
%Protocol: 25007 进阶
%------------------------------------------
c >> s:
 	int:8 AutoBuy	0非自动购买,1自动购买材料
s >> c:
  int:8 Result 返回码:0 失败,1 成功,2没有宠物,3没有进阶规则,4进阶丹不足,5铜钱不足,6元宝不足,7成长值不足
  int:8 Level  品阶
  int:16 FailTimes 失败次数
end

%------------------------------------------
%Protocol: 25008 成长进化
%------------------------------------------
c >> s:
 	int:8 AutoBuy	0非自动购买,1自动购买材料
 	int:8 AutoEvolve 0非自动进化, 1一键进化
s >> c:
  int:8 Result 返回码:0 失败,1 成功,2没有宠物,3传入参数错误,4没有进化规则,5达到进化最高级,6没有进化材料,7铜钱不足,8元宝不足
  int:16 当前成长值
  int:16 当前成长进度
  int:16 当前成长总进度
end

%------------------------------------------
%Protocol: 25009 提示资质
%------------------------------------------
c >> s:
 	int:8 AutoBuy	0非自动购买,1自动购买材料
 	int:8 AutoEvolve 0非自动提升, 1一键提升
s >> c:
  int:8 Result 返回码:0 失败,1 成功,2没有宠物,3传入参数错误,4没有进化规则,5达到进化最高级,6没有进化材料,7铜钱不足,8元宝不足
  int:16 当前资质
  int:16 当前提升进度
  int:16 当前提升总进度
end

%------------------------------------------
